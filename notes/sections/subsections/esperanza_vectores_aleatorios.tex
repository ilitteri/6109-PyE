        \item Si $X_{1}, \dots, X_{n}$ son independientes entonces
        \begin{equation*}
            E\left(\prod_{i=1}^{n}X_{i}\right) = \prod_{i=1}^{n}E(X_{i}) 
        \end{equation*}

            \begin{equation*}
                \begin{aligned}
                    E(X \cdot Y) &= \int_{-\infty}^{\infty}\int_{-\infty}^{\infty} xyf_{X}(x)f_{Y}(y) dx dy\\

                \end{aligned}
            \end{equation*}
        \end{proof}
