\begin{definition}[Varianza]
    Sea $X$ una $V.A.$, definimos la varianza de $X$ como
    \begin{equation*}
        Var(X) = E((X - E(X))^{2})
    \end{equation*}
\end{definition}

\begin{property}
    \begin{equation*}
        \begin{aligned}
            Var(X) &= E((X - E(X))^{2})\\
                   &= E(X^{2}) - 2XE(X) + E(X)^{2}\\
                   &= E(X^{2}) - 2\mu E(X) + \mu^{2}\\
                   &= E(X^{2}) - \mu^{2}\\
                   &= E(X^{2}) - E(X)^{2}
        \end{aligned}
    \end{equation*}
\end{property}
