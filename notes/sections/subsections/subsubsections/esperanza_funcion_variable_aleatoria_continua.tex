\begin{definition}[Esperanza de una función de una variable aleatoria continua]
    El valor de la esperanza de cualquier función $h(X)$ se calcula como
    \begin{equation*}
        E(h(X)) = \int_{-\infty}^{\infty} h(X) \cdot f_{X}(x) dx
    \end{equation*}
\end{definition}
