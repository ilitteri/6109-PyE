\begin{definition}[Esperanza parcial]
    \begin{equation*}
        E(X|X \in A) = \frac{E(X\mathbf{1}\{X \in A\})}{P(X \in A)}
    \end{equation*}
    Para la esperanza parcial uso la fórmula de esperanza (general) pero solo en la región que indeica la indicadora.
\end{definition}
