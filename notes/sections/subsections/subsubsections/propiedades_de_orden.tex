Sea $\mathbb{X} = X_{1}, \dots, X_{n}$ un vector aleatorio, $g: \mathbb{R}^{k} \rightarrow \mathbb{R}$, tenemos que
\begin{enumerate}
    \item Si $X > 0$ entonces $E(X)>0$.
    \item Si $g(X) > 0$ enttonces $E(g(X)) > 0$.
    \item Sea $h(X) > g(X)$ entonces $E(h(X)) > E(g(X))$.
    \item $E(|X|) \geq E(X)$.
    \item $E(|XY) \leq \sqrt{E(X^{2})E(Y^{2})}$.
\end{enumerate}
