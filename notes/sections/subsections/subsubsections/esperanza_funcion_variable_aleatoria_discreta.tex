\begin{definition}[Esperanza de una función de una variable aleatoria discreta]
    El valor de la esperanza de cualquier función $h(X)$ se calcula como
    \begin{equation*}
        E(h(X)) = \sum_{x \in R_{X}} h(x) \cdot p_{X}(x)
    \end{equation*}
\end{definition}
